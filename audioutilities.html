Audio Utilities

/// <summary> /// Play a sound effect once. /// </summary> /// <param name="audioSource">The AudioSource to play the sound effect on.</param> /// <param name="soundEffect">The AudioClip representing the sound effect to play.</param> /// <param name="volume">The volume at which to play the sound effect (default is 1.0).</param> public static void PlaySoundEffect(AudioSource audioSource, AudioClip soundEffect, float volume = 1.0f) { audioSource.PlayOneShot(soundEffect, volume); } /// <summary> /// Play background music. /// </summary> /// <param name="audioSource">The AudioSource to play the background music on.</param> /// <param name="music">The AudioClip representing the background music to play.</param> /// <param name="volume">The volume at which to play the background music (default is 1.0).</param> public static void PlayBackgroundMusic(AudioSource audioSource, AudioClip music, float volume = 1.0f) { audioSource.clip = music; audioSource.volume = volume; audioSource.loop = true; audioSource.Play(); } /// <summary> /// Stop audio playback. /// </summary> /// <param name="audioSource">The AudioSource to stop.</param> public static void StopAudio(AudioSource audioSource) { audioSource.Stop(); } /// <summary> /// Pause audio playback. /// </summary> /// <param name="audioSource">The AudioSource to pause.</param> public static void PauseAudio(AudioSource audioSource) { audioSource.Pause(); } /// <summary> /// Resume audio playback. /// </summary> /// <param name="audioSource">The AudioSource to resume.</param> public static void ResumeAudio(AudioSource audioSource) { audioSource.UnPause(); } /// <summary> /// Fade the volume of an AudioSource to a target value over a specified duration. /// </summary> /// <param name="audioSource">The AudioSource to fade.</param> /// <param name="targetVolume">The target volume to fade to.</param> /// <param name="fadeDuration">The duration of the fade in seconds.</param> public static System.Collections.IEnumerator FadeAudioVolume(AudioSource audioSource, float targetVolume, float fadeDuration) { float startVolume = audioSource.volume; float currentTime = 0; while (currentTime < fadeDuration) { audioSource.volume = Mathf.Lerp(startVolume, targetVolume, currentTime / fadeDuration); currentTime += Time.deltaTime; yield return null; } audioSource.volume = targetVolume; } /// <summary> /// Crossfade between two AudioSources over a specified duration. /// </summary> /// <param name="audioSource1">The first AudioSource to crossfade from.</param> /// <param name="audioSource2">The second AudioSource to crossfade to.</param> /// <param name="fadeDuration">The duration of the crossfade in seconds.</param> public static System.Collections.IEnumerator CrossfadeAudio(AudioSource audioSource1, AudioSource audioSource2, float fadeDuration) { float currentTime = 0; audioSource2.volume = 0; audioSource2.Play(); while (currentTime < fadeDuration) { audioSource1.volume = Mathf.Lerp(1, 0, currentTime / fadeDuration); audioSource2.volume = Mathf.Lerp(0, 1, currentTime / fadeDuration); currentTime += Time.deltaTime; yield return null; } audioSource1.Stop(); } /// <summary> /// Randomly pitch shift the audio source within a specified range. /// </summary> /// <param name="audioSource">The AudioSource to pitch shift.</param> /// <param name="minPitch">The minimum pitch value.</param> /// <param name="maxPitch">The maximum pitch value.</param> public static void RandomizePitch(AudioSource audioSource, float minPitch, float maxPitch) { audioSource.pitch = Random.Range(minPitch, maxPitch); } /// <summary> /// Play a random clip from an array of AudioClips on the given AudioSource. /// </summary> /// <param name="audioSource">The AudioSource to play the clip on.</param> /// <param name="clips">An array of AudioClips to choose from.</param> public static void PlayRandomClip(AudioSource audioSource, AudioClip[] clips) { if (clips.Length == 0) { Debug.LogWarning("No audio clips provided for random playback."); return; } int randomIndex = Random.Range(0, clips.Length); audioSource.clip = clips[randomIndex]; audioSource.Play(); } /// <summary> /// Loop an AudioSource's playback between two given time points. /// </summary> /// <param name="audioSource">The AudioSource to loop.</param> /// <param name="loopStart">The loop's start time in seconds.</param> /// <param name="loopEnd">The loop's end time in seconds.</param> public static void LoopAudioBetweenTimes(AudioSource audioSource, float loopStart, float loopEnd) { audioSource.time = loopStart; audioSource.SetScheduledEndTime(AudioSettings.dspTime + (loopEnd - loopStart)); audioSource.PlayScheduled(AudioSettings.dspTime + 0.1); } /// <summary> /// Mute or unmute an AudioSource's volume. /// </summary> /// <param name="audioSource">The AudioSource to mute/unmute.</param> /// <param name="mute">True to mute, false to unmute.</param> public static void MuteAudio(AudioSource audioSource, bool mute) { audioSource.mute = mute; } /// <summary> /// Set the pitch of an AudioSource to play in reverse. /// </summary> /// <param name="audioSource">The AudioSource to reverse.</param> public static void PlayReverse(AudioSource audioSource) { audioSource.pitch = -1; audioSource.Play(); } /// <summary> /// Change the pitch of an AudioSource. /// </summary> /// <param name="audioSource">The AudioSource to adjust the pitch of.</param> /// <param name="newPitch">The new pitch value.</param> public static void ChangePitch(AudioSource audioSource, float newPitch) { audioSource.pitch = newPitch; } /// <summary> /// Crossfade between two AudioSources over a specified duration. /// </summary> /// <param name="sourceA">The first AudioSource to crossfade from.</param> /// <param name="sourceB">The second AudioSource to crossfade to.</param> /// <param name="duration">The duration of the crossfade in seconds.</param> public static void Crossfade(AudioSource sourceA, AudioSource sourceB, float duration) { sourceB.volume = 0.0f; sourceB.Play(); float startTime = Time.time; float endTime = startTime + duration; while (Time.time < endTime) { float t = (Time.time - startTime) / duration; sourceA.volume = 1 - t; sourceB.volume = t; } sourceA.volume = 0.0f; } /// <summary> /// Fade out the volume of an AudioSource over a specified duration. /// </summary> /// <param name="audioSource">The AudioSource to fade out.</param> /// <param name="duration">The duration of the fade out in seconds.</param> public static void FadeOut(AudioSource audioSource, float duration) { float startVolume = audioSource.volume; float startTime = Time.time; float endTime = startTime + duration; while (Time.time < endTime) { float t = (Time.time - startTime) / duration; audioSource.volume = Mathf.Lerp(startVolume, 0.0f, t); } audioSource.volume = 0.0f; } /// <summary> /// Get the length of an AudioClip in seconds. /// </summary> /// <param name="clip">The AudioClip to get the length of.</param> /// <returns>The length of the AudioClip in seconds.</returns> public static float GetClipLength(AudioClip clip) { return clip.length; } /// <summary> /// Play an AudioClip on an AudioSource. /// </summary> /// <param name="audioSource">The AudioSource to play the AudioClip on.</param> /// <param name="audioClip">The AudioClip to play.</param> public static void PlayClip(AudioSource audioSource, AudioClip audioClip) { audioSource.clip = audioClip; audioSource.Play(); } /// <summary> /// Set the volume of an AudioSource. /// </summary> /// <param name="audioSource">The AudioSource to adjust the volume of.</param> /// <param name="volume">The new volume value (0.0 to 1.0).</param> public static void SetVolume(AudioSource audioSource, float volume) { audioSource.volume = Mathf.Clamp01(volume); } /// <summary> /// Set the loop flag on an AudioSource. /// </summary> /// <param name="audioSource">The AudioSource to set loop mode for.</param> /// <param name="loop">True to enable looping, false to disable looping.</param> public static void SetLoop(AudioSource audioSource, bool loop) { audioSource.loop = loop; } /// <summary> /// Check if an AudioSource is currently playing. /// </summary> /// <param name="audioSource">The AudioSource to check.</param> /// <returns>True if the AudioSource is playing, false otherwise.</returns> public static bool IsPlaying(AudioSource audioSource) { return audioSource.isPlaying; } /// <summary> /// Get the current time position within the AudioClip. /// </summary> /// <param name="audioSource">The AudioSource to get the time position from.</param> /// <returns>The current time position in seconds.</returns> public static float GetTimePosition(AudioSource audioSource) { return audioSource.time; } /// <summary> /// Set the time position within the AudioClip. /// </summary> /// <param name="audioSource">The AudioSource to set the time position on.</param> /// <param name="time">The new time position in seconds.</param> public static void SetTimePosition(AudioSource audioSource, float time) { audioSource.time = Mathf.Clamp(time, 0f, audioSource.clip.length); } /// <summary> /// Play an audio clip on an AudioSource after a specified delay. /// </summary> /// <param name="audioSource">The AudioSource to play the audio clip on.</param> /// <param name="audioClip">The audio clip to play.</param> /// <param name="delay">The delay before playing the audio clip in seconds.</param> public static IEnumerator PlayDelayed(AudioSource audioSource, AudioClip audioClip, float delay) { yield return new WaitForSeconds(delay); audioSource.clip = audioClip; audioSource.Play(); } /// <summary> /// Set the pitch of an AudioSource. /// </summary> /// <param name="audioSource">The AudioSource to adjust the pitch of.</param> /// <param name="pitch">The new pitch value (0.0 to 3.0).</param> public static void SetPitch(AudioSource audioSource, float pitch) { audioSource.pitch = Mathf.Clamp(pitch, 0f, 3f); } /// <summary> /// Fade in the volume of an AudioSource gradually. /// </summary> /// <param name="audioSource">The AudioSource to fade in.</param> /// <param name="duration">The duration of the fade-in in seconds.</param> public static IEnumerator FadeIn(AudioSource audioSource, float duration) { float startVolume = 0; audioSource.volume = 0; float startTime = Time.time; while (audioSource.volume < 1) { float elapsedTime = Time.time - startTime; audioSource.volume = Mathf.Lerp(startVolume, 1, elapsedTime / duration); yield return null; } audioSource.volume = 1; } /// <summary> /// Start playing an AudioSource. /// </summary> /// <param name="audioSource">The AudioSource used to play the audio.</param> public static void PlayAudio(AudioSource audioSource) { if (audioSource != null) { if (audioSource.clip != null) { audioSource.Play(); } else { Debug.LogWarning("Audio clip is null!"); } } else { Debug.LogWarning("Audio source is null!"); } } /// <summary> /// Set an AudioSource to 3D audio mode. /// </summary> /// <param name="audioSource">The AudioSource to set to 3D audio mode.</param> public static void SetAudio3D(AudioSource audioSource) { if (audioSource != null) { audioSource.spatialBlend = 1.0f; // Set spatialBlend to 1.0 for 3D audio. } } /// <summary> /// Set an AudioSource to 2D audio mode. /// </summary> /// <param name="audioSource">The AudioSource to set to 2D audio mode.</param> public static void SetAudio2D(AudioSource audioSource) { if (audioSource != null) { audioSource.spatialBlend = 0.0f; // Set spatialBlend to 0.0 for 2D audio. } }
